function animateValue(id, start, end, duration) {
    const element = document.getElementById(id);
    const increment = (end - start) / (duration / 16);
    let current = start;

    const timer = setInterval(() => {
        current += increment;
        if (current >= end) {
            current = end;
            clearInterval(timer);
        }
        element.textContent = Math.floor(current) + (id.includes('rate') ? '%' : '');
    }, 16);
}
